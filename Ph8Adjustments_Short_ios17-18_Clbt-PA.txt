SELECT
	ZUNMANAGEDADJUSTMENT.ZADJUSTMENTTIMESTAMP,
	ZASSET.ZDATECREATED,
	ZASSET.ZADDEDDATE,
	ZASSET.ZADJUSTMENTSSTATE,		  
	ZADDITIONALASSETATTRIBUTES.ZEDITORBUNDLEID,
	ZUNMANAGEDADJUSTMENT.ZEDITORLOCALIZEDNAME,
	ZUNMANAGEDADJUSTMENT.ZADJUSTMENTFORMATIDENTIFIER,
	ZASSET.ZDIRECTORY,
	ZASSET.ZFILENAME,
	ZADDITIONALASSETATTRIBUTES.ZORIGINALFILENAME,
	ZCLOUDMASTER.ZORIGINALFILENAME,
	ZASSET.Z_PK,
	ZADDITIONALASSETATTRIBUTES.Z_PK
FROM ZASSET
	LEFT JOIN ZADDITIONALASSETATTRIBUTES ON ZADDITIONALASSETATTRIBUTES.Z_PK = ZASSET.ZADDITIONALATTRIBUTES
	LEFT JOIN ZUNMANAGEDADJUSTMENT ON ZADDITIONALASSETATTRIBUTES.ZUNMANAGEDADJUSTMENT = ZUNMANAGEDADJUSTMENT.Z_PK
	LEFT JOIN ZCLOUDMASTER ON ZASSET.ZMASTER = ZCLOUDMASTER.Z_PK	
WHERE ZASSET.ZADJUSTMENTSSTATE > 0
ORDER BY ZUNMANAGEDADJUSTMENT.ZADJUSTMENTTIMESTAMP

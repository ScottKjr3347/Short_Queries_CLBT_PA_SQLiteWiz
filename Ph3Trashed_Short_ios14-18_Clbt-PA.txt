SELECT
ZASSET.ZTRASHEDDATE,
ZASSET.ZTRASHEDSTATE,
ZASSET.ZTRASHEDBYPARTICIPANT,
ZASSET.ZDIRECTORY,
ZASSET.ZFILENAME,
ZADDITIONALASSETATTRIBUTES.ZORIGINALFILENAME,
ZCLOUDMASTER.ZORIGINALFILENAME,
ZASSET.Z_PK,
ZADDITIONALASSETATTRIBUTES.Z_PK
FROM ZASSET ZASSET
    LEFT JOIN ZADDITIONALASSETATTRIBUTES ON ZADDITIONALASSETATTRIBUTES.Z_PK = ZASSET.ZADDITIONALATTRIBUTES
    LEFT JOIN ZCLOUDMASTER ZCLOUDMASTER ON ZASSET.ZMASTER = ZCLOUDMASTER.Z_PK
WHERE ZASSET.ZTRASHEDSTATE = 1
ORDER BY ZASSET.ZTRASHEDSTATE

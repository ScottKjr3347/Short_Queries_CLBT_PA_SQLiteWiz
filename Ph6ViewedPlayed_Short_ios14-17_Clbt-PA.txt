SELECT
ZASSET.ZMODIFICATIONDATE,
ZASSET.ZANALYSISSTATEMODIFICATIONDATE,
ZASSET.ZADDEDDATE,
ZASSET.ZDIRECTORY,
ZASSET.ZFILENAME,
ZADDITIONALASSETATTRIBUTES.ZORIGINALFILENAME,
ZADDITIONALASSETATTRIBUTES.ZEXIFTIMESTAMPSTRING,
ZADDITIONALASSETATTRIBUTES.ZPENDINGVIEWCOUNT,
ZADDITIONALASSETATTRIBUTES.ZVIEWCOUNT,
ZADDITIONALASSETATTRIBUTES.ZPENDINGPLAYCOUNT,
ZADDITIONALASSETATTRIBUTES.ZPLAYCOUNT, 
ZASSET.Z_PK,
ZADDITIONALASSETATTRIBUTES.Z_PK
FROM ZASSET
    LEFT JOIN ZADDITIONALASSETATTRIBUTES ON ZADDITIONALASSETATTRIBUTES.Z_PK = ZASSET.ZADDITIONALATTRIBUTES
WHERE (ZADDITIONALASSETATTRIBUTES.ZPENDINGVIEWCOUNT > 0) OR (ZADDITIONALASSETATTRIBUTES.ZVIEWCOUNT > 0) OR (ZADDITIONALASSETATTRIBUTES.ZPENDINGPLAYCOUNT > 0) OR (ZADDITIONALASSETATTRIBUTES.ZPLAYCOUNT > 0)
ORDER BY zAsset.ZMODIFICATIONDATE

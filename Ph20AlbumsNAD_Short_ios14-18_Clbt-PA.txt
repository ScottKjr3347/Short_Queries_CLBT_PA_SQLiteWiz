SELECT
ZGENERICALBUM.ZCREATIONDATE,
ZGENERICALBUM.ZSTARTDATE,
ZGENERICALBUM.ZENDDATE,
ZGENERICALBUM.ZKIND,
ZGENERICALBUM.ZTITLE, 
ZGENERICALBUM.ZCACHEDPHOTOSCOUNT,
ZGENERICALBUM.ZCACHEDVIDEOSCOUNT,
ZGENERICALBUM.ZCACHEDCOUNT,
ZGENERICALBUM.ZTRASHEDSTATE,
ZGENERICALBUM.ZUUID,
ZGENERICALBUM.ZCLOUDGUID
FROM ZGENERICALBUM
ORDER BY ZGENERICALBUM.ZCREATIONDATE

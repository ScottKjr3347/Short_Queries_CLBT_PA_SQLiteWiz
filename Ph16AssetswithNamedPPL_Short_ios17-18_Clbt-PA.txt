SELECT
ZASSET.ZDATECREATED,
ZASSET.ZDIRECTORY,
ZASSET.ZFILENAME,
ZADDITIONALASSETATTRIBUTES.ZORIGINALFILENAME,
ZPERSON.ZDISPLAYNAME,
ZPERSON.ZFULLNAME,
ZASSET.Z_PK,
ZDETECTEDFACE.Z_PK,
ZDETECTEDFACEPRINT.ZFACE,
ZFACECROP.ZFACE,
ZPERSON.Z_PK
FROM ZASSET
	LEFT JOIN ZADDITIONALASSETATTRIBUTES ON ZADDITIONALASSETATTRIBUTES.Z_PK = ZASSET.ZADDITIONALATTRIBUTES
	LEFT JOIN ZDETECTEDFACE ON ZASSET.Z_PK = ZDETECTEDFACE.ZASSETFORFACE
	LEFT JOIN ZPERSON ON ZPERSON.Z_PK = ZDETECTEDFACE.ZPERSONFORFACE
	LEFT JOIN ZDETECTEDFACEPRINT ON ZDETECTEDFACEPRINT.ZFACE = ZDETECTEDFACE.Z_PK
	LEFT JOIN ZFACECROP ON ZPERSON.Z_PK = ZFACECROP.ZPERSON
WHERE ZPERSON.ZDISPLAYNAME > 0 or ZPERSON.ZFULLNAME > 0
ORDER BY ZASSET.ZDATECREATED
